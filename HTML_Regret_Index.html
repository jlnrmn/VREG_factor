from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split

from dateutil.relativedelta import relativedelta


def compute_reg(date, PERMNO, df = df_reg, df_stocks = stocks):
    
    df_stocks = df_stocks.loc[df_stocks['PERMNO'] == PERMNO]
    df_stocks = df_stocks.loc[pd.to_datetime(df_stocks.index) < date]
    df_stocks = df_stocks.loc[pd.to_datetime(df_stocks.index) >= (pd.to_datetime(date) - relativedelta(months=60))]
    
    temp_df = pd.merge(left=df, right=df_stocks,
                      left_on=df.index, right_on=df_stocks.index).rename(columns={'key_0':'date'}).drop(columns={'date'})
    
    temp_df = temp_df.dropna()
    
    X = temp_df.iloc[:,:-1].values
    y = temp_df.iloc[:,-1].values
    
    try:
    
        X_train, X_test, y_train, y_test = train_test_split(X, y, test_size = 1/3, random_state = 0)

        regressor = LinearRegression()
        regressor.fit(X_train,y_train)
        y_pred = regressor.predict(X_test)

        beta_vreg = regressor.coef_[1] 
    
    except:
        beta_vreg = np.nan
        
    return date, PERMNO, beta_vreg



def get_beta_reg(df_stocks = stocks):
    
    list_beta = []
    
    for PERMNO in df_stocks['PERMNO'].unique(): 
        for date in df_stocks.index.unique()[60:]:
       
            list_beta.append(pd.DataFrame([compute_reg(date, PERMNO)]))
    
    df = pd.concat([x for x in list_beta])
    df.columns = ['date', 'PERMNO', 'beta_vreg']
    return df

FVREG¶

Use the mimicking factor methodology as outlined in Fama and French (1993).
Each month from January 1991 (1995) to December 2019, construct 6 (2x3) portfolios based on the intersections of 2 portfolios formed on size (market equity) and 3 portfolios formed on VREG.
The monthly size breakpoint is the median NYSE market equity (ME), and the monthly VREG breakpoints are the 30th and 70th NYSE VREG percentiles.
